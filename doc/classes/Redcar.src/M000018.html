  def self.menu(name, &block)
    @menus ||= {}
    if @menus[name]
      menu = self.common(name, block)
    else
      menu = self.common(name, block)
      menuitem = Gtk::MenuItem.new(name)
      menuitem.submenu = menu
      menuitem.show
      Redcar.menubar.append(menuitem)
    end
  end
