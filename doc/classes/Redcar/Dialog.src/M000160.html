    def self.button_convert(buttoninfo)
      case buttoninfo.to_s.downcase
      when "find"
        Gtk::Stock::FIND
      when "replace"
        Gtk::Stock::FIND_AND_REPLACE
      when "ok"
        Gtk::Stock::OK
      when "save"
        Gtk::Stock::SAVE
      else
        buttoninfo
      end
    end
