    def self.set_sensitive_widget(gtkw, name)
      unless defined? @@sensitivities and @@sensitivities[name]
        puts "Trying to make a widget sensitive to a sensitivity that"+
          " doesn't exist! (#{name})"
        raise ArgumentError
      end
      @@sensitivities[name] << gtkw
      gtkw.sensitive = false
    end
