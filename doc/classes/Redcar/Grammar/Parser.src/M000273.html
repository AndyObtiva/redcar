      def parse_line(line, line_num)
        unless line_num < @lines.length
          raise ArgumentError, "cannot parse line that does not already exist"
        end
        current    = @lines[line_num]
        new_scopes = []
        
        puts "current: #{current.inspect}"
        start_scope = line_start(line_num)
        puts "start_scope: #{start_scope.inspect}"
        end_scope   = line_end(line_num)
        puts "end_scope: #{end_scope.inspect}"
        
        if start_scope.respond_to? :grammar
          active_grammar = start_scope.grammar
        else
          active_grammar = grammar_for_scope(start_scope)
        end
        puts "active_grammar: #{active_grammar.inspect}"
        
        pos = 0
        current_scope = start_scope
        
        while pos < line.length
          puts
          rest_line = line[pos..-1]
          p rest_line
          pps = active_grammar.possible_patterns(current_scope)
          puts "current_scope: #{current_scope.inspect}"
          puts "possible patterns:"
          pps.each {|p| p p}
          new_scope_markers = []
          if current_scope.is_a? ScopeMarker
            puts "closing regexp: #{current_scope.closing_regexp}"
            if md = Regexp.new(current_scope.closing_regexp).match(rest_line)
              p :closing_matches
              sm = CloseScopeMarker.new(md.pre_match.length+pos,
                                        md.to_s.length+md.pre_match.length+pos,
                                        current_scope.pattern, active_grammar,
                                        md, nil)
              new_scope_markers << sm
            end
          end
          puts "matches:"
          pps.each do |pattern|
            case pattern
            when SinglePattern
              md = Regexp.new(pattern.match).match(rest_line)
              if md
                puts "matched SinglePattern"
                p md.to_s
                p md.captures
                p md.pre_match.length+pos
                p md.to_s.length+md.pre_match.length+pos
                sm = SingleScopeMarker.new(md.pre_match.length+pos,
                                           md.to_s.length+md.pre_match.length+pos,
                                           pattern, active_grammar,
                                           md, current_scope)
                new_scope_markers << sm
              end
            when DoublePattern
              md = Regexp.new(pattern.begin).match(rest_line)
              if md
                puts "matched DoublePattern"
                p md.to_s
                p md.captures
                p md.pre_match.length+pos
                p md.to_s.length+md.pre_match.length+pos
                sm = OpenScopeMarker.new(md.pre_match.length+pos,
                                         md.to_s.length+md.pre_match.length+pos,
                                         pattern, active_grammar,
                                         md, current_scope)
                sm.previous_scope = current_scope
                sm.closing_regexp = build_closing_regexp(pattern, md)
                new_scope_markers << sm
              end
            end
          end
          if new_scope_markers.length > 0
            first_new_scope = new_scope_markers.sort_by {|sm| sm.from }[0]
            puts "first_new_scope: #{first_new_scope.pattern.inspect}"
            new_scopes << first_new_scope
            pos = first_new_scope.to
            if first_new_scope.pattern.is_a? DoublePattern
              if first_new_scope.is_a? OpenScopeMarker
                current_scope = first_new_scope
              elsif first_new_scope.is_a? CloseScopeMarker
                puts "resetting scope to previous"
                current_scope = current_scope.previous_scope
              else
                raise ArgumentError, "unknown scope marker"
              end
            end
          else
            pos = line.length + 1
          end
        end
        p new_scopes.map{|sm| "#{sm.from}-#{sm.to}:#{sm.pattern.name} (#{sm.matchdata.to_s})"}
        
        @lines[line_num] = new_scopes
        new_scopes.last == end_scope
      end
