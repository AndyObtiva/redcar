    def get_lines(selector)
      if selector.is_a? Range
        st = selector.begin
        en = selector.end
        if st < 0
          nst = @buffer.line_count+st
        else
          nst = st
        end
        if en < 0
          nen = @buffer.line_count+en
        else
          nen = en
        end
        a = [nst, nen].sort
        selector = a[0]..a[1]
      end
      selector.map{|num| get_line(num)}
    end
