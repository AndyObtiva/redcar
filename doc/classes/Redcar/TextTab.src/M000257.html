      def find_next(str)
        rest = self.contents[(cursor_offset+1)..-1]
        return nil unless rest
        if md = rest.match(/#{str}/)
          p md.offset(0).map{|e| e+cursor_offset+1}
          select(*(md.offset(0).map{|e| e+cursor_offset+1}))
          true
        else
          nil
        end
      end
