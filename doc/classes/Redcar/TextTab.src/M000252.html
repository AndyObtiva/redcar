      def []=(obj, str)
        case obj.class.to_s
        when "Fixnum", "Bignum", "Integer"
          old = self[obj]
          delete(obj, obj+1)
          insert(obj, str)
        when "Range"
          delete(obj.first, obj.last+1)
          insert(obj.first, str)
        end
      end
