    def above_offset(offset)
      above_line_num = [iter(offset).line-1, 0].max
      return 0 if above_line_num == 0
      [
       @buffer.get_iter_at_line(above_line_num).offset + 
         [iter(offset).line_offset, get_line(above_line_num).length-1].min,
       0
      ].max
    end
