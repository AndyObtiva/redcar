    def iter(thing)
      case thing
      when Integer
        thing = [0, thing].max
        thing = [length, thing].min
        @buffer.get_iter_at_offset(thing)
      when Gtk::TextMark
        @buffer.get_iter_at_mark(thing)
      when Gtk::TextIter
        thing
      when TextLoc
        line_start = @buffer.get_iter_at_line(thing.line)
        iter(line_start.offset+thing.offset)
      end
    end
