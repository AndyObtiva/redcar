    def tab_position=(position)
      @tab_position = position
      case position
      when :bottom
        @notebook.set_tab_pos(Gtk::POS_BOTTOM)
      when :left
        @notebook.set_tab_pos(Gtk::POS_LEFT)
      when :right
        @notebook.set_tab_pos(Gtk::POS_RIGHT)
      else
        @notebook.set_tab_pos(Gtk::POS_TOP)
      end
    end
