    def self.plist_to_xml(plist)
      doc = REXML::Document.new
      xml_el = doc.add_element "plist", {"version" => "1.0"}
      plist.each do |el|
        plist_to_xml1(el, xml_el)
      end
      str = ""
      doc.write(str, 0)
      dt="<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n<!DOCTYPE plist PUBLIC \"-//Apple Computer//DTD PLIST 1.0//EN\" \"http://www.apple.com/DTDs/PropertyList-1.0.dtd\">\n"
      (dt+str).gsub("\'", "\"")
    end
