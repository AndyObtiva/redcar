    def self.parse(str)
      return str if str.is_a? KeyBinding
      str = str.strip
      if str.strip.include? " "
        str_mods, str_keyname = str.strip.split(" ")
        mods = str_mods.split('-').map do |sm|
          case sm.downcase
          when "alt", "a"
            :alt
          when "control", "ctrl", "c"
            :control
          when "shift", "sh", "s"
            :shift
          when "caps", "lock", "capslock", "cl"
            :caps
          when "super", "sup", "spr", "sp"
            :super
          end
        end
        self.new(mods, str_keyname)
      else
        self.new([], str)
      end
    end
