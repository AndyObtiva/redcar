      def inspect
        if self.pattern.is_a? SinglePattern
          hanging = ""
        else
          if self.end
            hanging = " closed"
          else
            hanging = " hanging"
          end
        end
        startstr = "(#{start.line},#{start.offset})-"
        if self.end
          endstr = "(#{self.end.line},#{self.end.offset})"
        else
          endstr = "?"
        end
        "<scope(#{self.object_id*-1%1000}):"+(self.name||"noname")+" #{startstr}#{endstr} #{hanging}>"
      end
