      def initialize(options)
        options = process_params(options, 
                                 { :grammar => MANDATORY,
                                   :start => MANDATORY,
                                   :start_marker => nil,
                                   :end_marker => nil,
                                   :pattern => MANDATORY,
                                   :end => nil,
                                   :parent => nil,
                                   :closing_regexp => nil
                                 })
        self.pattern = options[:pattern]
        self.grammar = options[:grammar]
        raise ArgumentError, "not a TextLocation" unless options[:start].is_a? TextLoc
        self.start = options[:start]
        if options[:end]
          raise ArgumentError, "not a TextLoc" unless options[:end].is_a? TextLoc
        end
        self.end   = options[:end]
        self.children = []
        self.parent = options[:parent]
        self.closing_regexp = options[:closing_regexp]
      end
