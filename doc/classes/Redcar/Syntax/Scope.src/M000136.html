      def self.common_ancestor1(s1, s2, distance=0)
        if s1 == s2
          return [s1, distance]
        else
          if s1.parent
            c1, d1 = self.common_ancestor1(s1.parent, s2, distance+1)
          else
            c1 = nil
          end
          if s2.parent
            c2, d2 = self.common_ancestor1(s1, s2.parent, distance+1)
          else
            c2 = nil
          end
          if c1 and not c2
            return c1, d1
          elsif c2 and not c1
            return c2, d2
          elsif c1 and c2
            if d1 <= d2
              return c1, d1
            else
              return c2, d2
            end
          else
            return nil, distance
          end
        end
      end
