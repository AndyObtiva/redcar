      def clear_after(textloc)
        if self.end > textloc
          self.end = nil
          self.end_marker = nil
        end
        children.each do |cs|
          if cs.start > textloc
            children.delete(cs)
          else
            cs.clear_after(textloc)
          end
        end
      end
