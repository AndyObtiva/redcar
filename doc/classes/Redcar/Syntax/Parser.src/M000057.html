      def indent(line_num)
        if line_num == 0
          0
        else
          ind = 0
          prev = indent(line_num-1)
          if @fold_counts[line_num-1] > 0
            ind = prev + 1
          elsif @fold_counts[line_num-1] < 0
            ind = prev
          elsif @fold_counts[line_num-1] == 0
            ind = prev
          end
          if @fold_counts[line_num] < 0
            ind - 1
          else
            ind
          end
        end
      end
