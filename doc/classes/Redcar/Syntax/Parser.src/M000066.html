      def clear_after(line_num)
        @scope_tree.clear_after(TextLoc.new(line_num, 0))
        if line_num == 0
          @text = []
          @fold_counts = []
        else
          @text = @text[0..line_num-1]
          @fold_counts[0..line_num-1]
        end
      end
