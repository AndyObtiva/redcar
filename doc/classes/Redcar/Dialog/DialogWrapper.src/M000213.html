      def show(args={})
        args = process_params(args, 
                              { :modal => false,
                                :block => false })
        if args[:modal]
          @dialog.modal = true
        end
        if args[:block]
          @dialog.run
        else
          @dialog.show_all
        end
        yield if block_given?
      end
