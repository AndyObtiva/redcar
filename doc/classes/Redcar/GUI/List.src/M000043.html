      def initialize(options={})
        options = process_params(options,
                                 { :type => String,
                                   :heading => nil,
                                   :multiple_select => false})
        upoptions = {}
        upoptions[:columns] = [{ :type => options[:type], 
                                 :heading => options[:heading] }]
        upoptions[:multiple_select] = options[:multiple_select]
        super(upoptions)
      end
