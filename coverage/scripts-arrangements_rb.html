201         if arrangement.keys.include? "types" # is a pane

217         if arrangement.keys.include? "types"
218           pane.tab_angle    = (arrangement["tab_angle"].intern || :horizontal)
219           pane.tab_position = (arrangement["tab_position"].intern || :top)

226             apply_arrangement1(arrangement["left"], pane1)
